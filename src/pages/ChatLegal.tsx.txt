import { useState, useRef, useEffect } from "react";

export default function ChatLegal() {
  const [moduloActivo, setModuloActivo] = useState("Monitor de Riesgo (Arg-Ven)");
  const [inputTexto, setInputTexto] = useState("");
  const [mensajes, setMensajes] = useState<{ texto: string; remitente: "usuario" | "ia" }[]>([]);
  const [isTyping, setIsTyping] = useState(false);
  const chatEndRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    chatEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [mensajes, isTyping]);

  const cambiarModulo = (nombreModulo: string) => {
    setModuloActivo(nombreModulo);
    setMensajes([]);
  };

  const enviarMensaje = () => {
    if (!inputTexto.trim()) return;

    setMensajes((prev) => [...prev, { texto: inputTexto, remitente: "usuario" }]);
    setInputTexto("");
    setIsTyping(true);

    setTimeout(() => {
      setIsTyping(false);
      setMensajes((prev) => [
        ...prev,
        {
          texto: `Simulación de sistema: Recibí sus datos en el departamento de ${moduloActivo}. La interfaz de React Zero-Avatar está operativa y lista para la IA.`,
          remitente: "ia",
        },
      ]);
    }, 1500);
  };

  return (
    <div className="flex h-screen w-screen overflow-hidden text-gray-200 font-sans bg-[#060b1a] m-0 p-0 absolute top-0 left-0 z-50">
      <aside className="w-64 bg-[#030712] flex-col border-r border-gray-800 hidden md:flex">
        <div className="p-6">
          <img src="/logo.png" alt="LAP Global Logo" className="w-20 h-24 mx-auto mb-3 object-contain drop-shadow-[0_0_15px_rgba(197,160,89,0.4)]" />
          <h2 className="text-center text-[11px] uppercase tracking-widest text-[#c5a059] font-bold">LAP Global</h2>
        </div>
        <nav className="flex-1 overflow-y-auto px-3 space-y-1">
          <p className="text-[10px] text-gray-500 font-bold px-3 mb-2 uppercase">Centro de Inteligencia</p>
          <button onClick={() => cambiarModulo("Monitor de Riesgo (Arg-Ven)")} className={`w-full flex items-center p-3 rounded-lg text-sm transition-all ${moduloActivo === "Monitor de Riesgo (Arg-Ven)" ? "bg-[#1f2937] border-l-4 border-[#c5a059]" : "hover:bg-[#111827]"}`}>🌐 Monitor de Riesgo</button>
          <button onClick={() => cambiarModulo("Análisis Penal (Arg-Ven)")} className={`w-full flex items-center p-3 rounded-lg text-sm transition-all ${moduloActivo === "Análisis Penal (Arg-Ven)" ? "bg-[#1f2937] border-l-4 border-[#c5a059]" : "hover:bg-[#111827]"}`}>⚖️ Análisis Penal</button>
          <button onClick={() => cambiarModulo("Auditoría Documental")} className={`w-full flex items-center p-3 rounded-lg text-sm transition-all ${moduloActivo === "Auditoría Documental" ? "bg-[#1f2937] border-l-4 border-[#c5a059]" : "hover:bg-[#111827]"}`}>📄 Auditoría Documental</button>
        </nav>
      </aside>

      <main className="flex-1 flex flex-col relative">
        <header className="h-16 border-b border-gray-800 flex items-center justify-between px-6 bg-[#060b1a]/80 backdrop-blur-md sticky top-0 z-10">
          <h2 className="font-medium text-gray-100">{moduloActivo}</h2>
          <div className="flex gap-2 items-center">
             {/* Botón para volver a la página principal */}
            <a href="/" className="text-xs text-[#c5a059] border border-[#c5a059] px-3 py-1 rounded hover:bg-[#c5a059] hover:text-black transition">Volver al Inicio</a>
          </div>
        </header>

        <section className="flex-1 overflow-y-auto p-4 md:p-12 space-y-6">
          <div className="max-w-3xl mx-auto flex gap-4 items-start mb-4">
            <img src="/logo.png" className="w-10 h-12 object-contain" />
            <div className="space-y-4 mt-1">
              <p className="text-xl font-light text-gray-300">Inteligencia Artificial <strong>LAP Global</strong>.</p>
              <p className="text-gray-400 leading-relaxed">Módulo <span className="text-[#c5a059]">{moduloActivo}</span> activo. ¿Cuál es su consulta?</p>
            </div>
          </div>
          {mensajes.map((msg, index) => (
            <div key={index} className={`max-w-3xl mx-auto flex mt-2 ${msg.remitente === "usuario" ? "justify-end" : "justify-start"}`}>
              <div className={`p-4 rounded-3xl max-w-[90%] shadow-md ${msg.remitente === "usuario" ? "bg-[#2a303c] text-gray-100 rounded-tr-none border border-gray-700" : "bg-gray-800 text-gray-200 rounded-tl-none border-l-4 border-[#c5a059]"}`}>
                <p className="text-base">{msg.texto}</p>
              </div>
            </div>
          ))}
          {isTyping && (
             <div className="max-w-3xl mx-auto flex justify-start mt-2">
                <div className="text-[#c5a059] text-sm font-medium animate-pulse ml-2">Analizando...</div>
             </div>
          )}
          <div ref={chatEndRef} />
        </section>

        <footer className="p-4 md:pb-8">
          <div className="max-w-3xl mx-auto relative group">
            <div className="bg-[#1e2330] rounded-3xl border border-gray-700 p-2 pl-4 flex items-end gap-2 focus-within:border-[#c5a059] transition-all shadow-2xl">
              <textarea
                value={inputTexto}
                onChange={(e) => setInputTexto(e.target.value)}
                onKeyDown={(e) => { if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); enviarMensaje(); } }}
                placeholder="Escriba aquí los detalles del caso..."
                rows={1}
                className="w-full bg-transparent text-gray-100 py-3 outline-none text-base resize-none overflow-hidden"
              />
              <button onClick={enviarMensaje} className="bg-[#c5a059] text-black p-3 rounded-2xl mb-1 hover:bg-yellow-600 transition-all active:scale-95">
                Enviar
              </button>
            </div>
          </div>
        </footer>
      </main>
    </div>
  );
}
